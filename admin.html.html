    <script>
        function getUsers() { return JSON.parse(localStorage.getItem('dd_users') || '[]'); }
        function getAds() { return JSON.parse(localStorage.getItem('dd_ads') || '[]'); }
        function getBlocked() { return JSON.parse(localStorage.getItem('dd_blocked') || '[]'); }
        function saveBlocked(b) { localStorage.setItem('dd_blocked', JSON.stringify(b)); }
       
        function adminLogin() {
            if (document.getElementById('adminPassword').value === 'admin123') {
                document.getElementById('adminLogin').style.display = 'none';
                document.getElementById('adminPanel').style.display = 'block';
                loadData();
            } else alert('Нотўғри парол');
        }
       
        function loadData() {
            const users = getUsers();
            const ads = getAds();
            const blocked = getBlocked();
            document.getElementById('totalUsers').textContent = users.length;
            document.getElementById('totalAds').textContent = ads.length;
            document.getElementById('blockedUsers').textContent = blocked.length;
           
            let html = '';
            users.forEach(u => {
                const isBlocked = blocked.includes(u.phone);
                html += `<tr>
                    <td>${u.name || 'Имзо'}</td>
                    <td>${u.phone}</td>
                    <td><span class="badge ${isBlocked ? 'blocked' : 'active'}">${isBlocked ? '🚫 Блок' : '✅ Фаол'}</span></td>
                    <td>
                        ${isBlocked ?
                            `<button class="btn-block unblock" onclick="toggleBlock('${u.phone}', false)">🔓 Очиш</button>` :
                            `<button class="btn-block block" onclick="toggleBlock('${u.phone}', true)">🚫 Блоклаш</button>`
                        }
                    </td>
                </tr>`;
            });
            document.getElementById('usersTable').innerHTML = html || '<tr><td colspan="4" style="text-align:center;">Хеч ким йўқ</td></tr>';
        }
       
        function toggleBlock(phone, block) {
            let blocked = getBlocked();
            if (block) { if (!blocked.includes(phone)) blocked.push(phone); }
            else blocked = blocked.filter(p => p !== phone);
            saveBlocked(blocked);
            loadData();
        }
