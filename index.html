<!DOCTYPE html>
<html lang="uz">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>IshTop - Kunlik ish platformasi</title>
    <style>
        /* CSS stillari - эски файлингиздаги барча услубларни шу ерга кўчиринг */
        /* Қисқартириш учун бу ерда фақат асосийлари келтирилган */
        body { font-family: sans-serif; margin: 0; padding: 0; background: #f4f6f8; }
        .container { max-width: 480px; margin: 0 auto; background: white; min-height: 100vh; position: relative; }
        /* ... қолган барча услубларни шу ерга жойланг ... */
    </style>
    <!-- Supabase JS библиотекаси -->
    <script src="https://cdn.jsdelivr.net/npm/@supabase/supabase-js@2"></script>
</head>
<body>
    <div class="container" id="app">
        <!-- САҲИФАЛАР -->
        <!-- Кириш саҳифаси -->
        <div id="login-screen" class="screen active">
            <h2>Кириш</h2>
            <input type="text" id="phone" placeholder="Телефон рақам">
            <button onclick="handleLogin()">Кириш</button>
            <p>Ҳали рўйхатдан ўтмаганмисиз? <a href="#" onclick="showRegister()">Рўйхатдан ўтиш</a></p>
        </div>

        <!-- Рўйхатдан ўтиш саҳифаси -->
        <div id="register-screen" class="screen">
            <h2>Рўйхатдан ўтиш</h2>
            <input type="text" id="reg-name" placeholder="Исмингиз">
            <input type="text" id="reg-phone" placeholder="Телефон рақам">
            <button onclick="handleRegister()">Рўйхатдан ўтиш</button>
            <p>Аккаунтингиз борми? <a href="#" onclick="showLogin()">Кириш</a></p>
        </div>

        <!-- Асосий саҳифа (элонлар рўйхати) -->
        <div id="home-screen" class="screen">
            <div class="header">
                <h2>Иш элонлари</h2>
                <button onclick="logout()">Чиқиш</button>
            </div>
            <!-- Категориялар фильтри -->
            <div class="categories">
                <button class="category active" data-cat="all">Барчаси</button>
                <button class="category" data-cat="home">Уй ишлари</button>
                <button class="category" data-cat="field">Дала ишлари</button>
                <button class="category" data-cat="build">Қурилиш</button>
                <button class="category" data-cat="daily">Кунлик иш</button>
            </div>
            <div id="ads-list" class="ads-list">
                <!-- Элонлар динамик тўлдирилади -->
            </div>
            <button class="add-button" onclick="showAddAd()">+ Элон қўшиш</button>
        </div>

        <!-- Элон қўшиш саҳифаси -->
        <div id="add-ad-screen" class="screen">
            <h2>Янги элон</h2>
            <input type="text" id="ad-title" placeholder="Сарлавҳа">
            <textarea id="ad-desc" placeholder="Тавсиф"></textarea>
            <input type="text" id="ad-price" placeholder="Нарх">
            <select id="ad-category">
                <option value="home">Уй ишлари</option>
                <option value="field">Дала ишлари</option>
                <option value="build">Қурилиш</option>
                <option value="daily">Кунлик иш</option>
            </select>
            <input type="text" id="ad-location" placeholder="Жойлашган манзил">
            <label>
                <input type="checkbox" id="ad-tezkor"> Тезкор (10 SM)
            </label>
            <button onclick="submitAd()">Элонни жойлаш</button>
            <button onclick="goHome()">Орқага</button>
        </div>
    </div>

    <script>
        // 1. Supabase клиентини яратиш
        const SUPABASE_URL = window.SUPABASE_URL || '{{ SUPABASE_URL }}'; // Netlify env dan олинади
        const SUPABASE_ANON_KEY = window.SUPABASE_ANON_KEY || '{{ SUPABASE_ANON_KEY }}'; // Netlify env дан олинади

        const supabase = supabaseJs.createClient(SUPABASE_URL, SUPABASE_ANON_KEY);

        // 2. Умумий ўзгарувчилар
        let currentUser = null;
        let currentCategory = 'all';

        // 3. Саҳифаларни бошқариш
        function showScreen(screenId) {
            document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
            document.getElementById(screenId).classList.add('active');
        }

        window.showLogin = () => showScreen('login-screen');
        window.showRegister = () => showScreen('register-screen');
        window.goHome = () => { showScreen('home-screen'); loadAds(); };
        window.showAddAd = () => {
            if (!currentUser) {
                alert('Аввал киринг');
                showLogin();
                return;
            }
            showScreen('add-ad-screen');
        };

        // 4. Аутентификация
        window.handleLogin = async () => {
            const phone = document.getElementById('phone').value;
            // Демо логин: телефон рақам бўйича фойдаланувчини оламиз
            const { data: user, error } = await supabase
                .from('users')
                .select('*')
                .eq('phone', phone)
                .single();

            if (error || !user) {
                alert('Фойдаланувчи топилмади. Аввал рўйхатдан ўтинг.');
                return;
            }

            currentUser = user;
            localStorage.setItem('currentUser', JSON.stringify(user));
            goHome();
        };

        window.handleRegister = async () => {
            const name = document.getElementById('reg-name').value;
            const phone = document.getElementById('reg-phone').value;

            const { data: user, error } = await supabase
                .from('users')
                .insert([{ name, phone }])
                .select()
                .single();

            if (error) {
                alert('Хатолик: ' + error.message);
                return;
            }

            currentUser = user;
            localStorage.setItem('currentUser', JSON.stringify(user));
            goHome();
        };

        window.logout = () => {
            currentUser = null;
            localStorage.removeItem('currentUser');
            showLogin();
        };

        // Сессияни тиклаш
        const savedUser = localStorage.getItem('currentUser');
        if (savedUser) {
            currentUser = JSON.parse(savedUser);
            showScreen('home-screen');
        }

        // 5. Элонлар билан ишлаш
        async function loadAds() {
            let query = supabase.from('ads').select('*').order('created_at', { ascending: false });

            if (currentCategory !== 'all') {
                query = query.eq('category', currentCategory);
            }

            const { data: ads, error } = await query;

            if (error) {
                console.error('Элонларни олишда хатолик:', error);
                return;
            }

            const adsList = document.getElementById('ads-list');
            adsList.innerHTML = '';

            if (!ads || ads.length === 0) {
                adsList.innerHTML = '<p>Ҳозирча элонлар йўқ.</p>';
                return;
            }

            ads.forEach(ad => {
                const adCard = document.createElement('div');
                adCard.className = 'ad-card';
                adCard.innerHTML = `
                    <h3>${ad.title}</h3>
                    <p>${ad.description || ''}</p>
                    <p><strong>Нарх:</strong> ${ad.price || 'Келишилади'}</p>
                    <p><strong>Категория:</strong> ${ad.category}</p>
                    <p><strong>Жойлашган:</strong> ${ad.location || 'Кўрсатилмаган'}</p>
                    <p><strong>Телефон:</strong> ${ad.phone}</p>
                    <p><strong>Элон берди:</strong> ${ad.author}</p>
                    ${ad.tezkor ? '<span class="badge">⚡ Тезкор</span>' : ''}
                `;
                adsList.appendChild(adCard);
            });
        }

        window.submitAd = async () => {
            if (!currentUser) {
                alert('Аввал киринг');
                showLogin();
                return;
            }

            const title = document.getElementById('ad-title').value;
            const description = document.getElementById('ad-desc').value;
            const price = document.getElementById('ad-price').value;
            const category = document.getElementById('ad-category').value;
            const location = document.getElementById('ad-location').value;
            const tezkor = document.getElementById('ad-tezkor').checked;

            if (!title || !category) {
                alert('Сарлавҳа ва категория мажбурий!');
                return;
            }

            const { data, error } = await supabase
                .from('ads')
                .insert([{
                    title,
                    description,
                    price,
                    category,
                    location,
                    tezkor,
                    author: currentUser.name,
                    phone: currentUser.phone,
                    date: new Date().toLocaleDateString()
                }])
                .select();

            if (error) {
                alert('Элон қўшишда хатолик: ' + error.message);
                return;
            }

            // Формани тозалаш
            document.getElementById('ad-title').value = '';
            document.getElementById('ad-desc').value = '';
            document.getElementById('ad-price').value = '';
            document.getElementById('ad-location').value = '';
            document.getElementById('ad-tezkor').checked = false;

            alert('Элон муваффақиятли қўшилди!');
            goHome();
        };

        // Категория фильтри
        document.querySelectorAll('.category').forEach(btn => {
            btn.addEventListener('click', (e) => {
                document.querySelectorAll('.category').forEach(b => b.classList.remove('active'));
                e.target.classList.add('active');
                currentCategory = e.target.dataset.cat;
                loadAds();
            });
        });

        // Агар home экрани актив бўлса, элонларни юклаш
        const observer = new MutationObserver(() => {
            if (document.getElementById('home-screen').classList.contains('active')) {
                loadAds();
            }
        });
        observer.observe(document.getElementById('app'), { attributes: true, subtree: true });

        // Бошланғич юклаш
        if (currentUser) {
            loadAds();
        }
    </script>
</body>
</html>
